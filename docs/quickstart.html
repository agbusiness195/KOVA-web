Quickstart

SDK live on npm.

Add accountability to your AI agent in minutes. Every action is monitored and verifiable.

import { buildCovenant, verifyCovenant } from '@usekova/core';

const covenant = await buildCovenant({
  issuer:      { id: 'agent-001', publicKey: pubKey, role: 'issuer' },
  beneficiary: { id: 'user-042',  publicKey: userKey, role: 'beneficiary' },
  constraints: "forbid transfer over 1000; forbid access on '/private/**'",
  privateKey:  signingKey
});

const valid = verifyCovenant(covenant);
